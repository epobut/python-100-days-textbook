Поурочные материалы

День 1: Установка и первый скрипт
# hello.py — первое приложение
print("Привет, Python!")  # вывод строки
День 2: Переменные и типы
a = 10            # int
b = 3.14          # float
name = "Алиса"     # str
is_ok = True      # bool
print(type(a), type(b), type(name), type(is_ok))
День 3: Ввод/вывод, форматирование
name = input("Имя: ")
age = int(input("Возраст: "))
print(f"Привет, {name}! Тебе {age} лет.")
День 4: Операторы
x, y = 7, 3
print(x + y, x - y, x * y, x / y)
print(x // y, x % y, x ** y)
День 5: Условия
score = int(input("Баллы: "))
print("A" if score >= 90 else "B" if score >= 75 else "C" if score >= 60 else "D")
День 6: while
n = 3
while n > 0:
    print(n)
    n -= 1
print("Готово")
День 7: for и range
for i in range(5):
    print(i)
День 8: Строки
s = "Программирование"
print(s[0], s[-1], s[0:6])
print(s.lower(), s.upper())
День 9: Списки
nums = [1,2,3]
nums.append(4)
squares = [x*x for x in nums]
print(nums, squares)
День 10: Кортежи и множества
t = (1,2,3)
s = {1,2,2,3}
print(t, s)
День 11: Словари
d = {"имя":"Алиса","возраст":20}
print(d.get("рост","нет данных"))
d["город"] = "Казань"
for k,v in d.items():
    print(k, v)
День 12: Функции
def greet(name: str = "мир") -> str:
    return f"Привет, {name}!"
print(greet(), greet("Алиса"))
День 13: Модули
import math
from datetime import date
print(round(math.pi, 3), date.today())
День 14: Ошибки
def safe_div(a,b):
    try:
        return a/b
    except ZeroDivisionError:
        return "Деление на ноль!"
print(safe_div(10,2), safe_div(5,0))
День 15: Файлы
text = "Строка\nещё строка"
with open("demo.txt","w", encoding="utf-8") as f:
    f.write(text)
with open("demo.txt","r", encoding="utf-8") as f:
    print(f.read())
